//----- Function for Random Warps ----------------------------
function	script	randomw	{
	set @rdm,rand(3);
	if(@rdm==1)goto rdm1;
	if(@rdm==2)goto rdm2;
 	warp "lhz_cube",67,193; end;
	rdm1: warp "lhz_cube",66,136; end;
	rdm2: warp "lhz_cube",66,74; end;
}
