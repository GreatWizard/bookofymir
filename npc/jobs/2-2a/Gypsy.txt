valkyrie,53,56,3	script	Gypsie#Valkyrie	101,{
	if (ADVJOB == 0 || Upper != 1) {
		set .@karma_d,rand(1,10);
		if (.@karma_d > 4) {
			mes "[Gypsie]";
			mes "Félicitations.";
			mes "Honneur aux Guerriers!";
			close;
		}
		mes "[Gypsie]";
		mes "Un pas à gauche,";
		mes "un pas à droite~!";
		mes "Et balance...!";
		mes "Danser est plus qu'un simple passe-temps. Pour moi, c'est un mode de vie~";
		close;
	}
	else if (ADVJOB == Job_Gypsy && Class == Job_Archer_High && JobLevel > 39) {
		mes "[Gypsie]";
		mes "La terre de Rune-Midgard a besoin de femmes talentueuses pour changer subtilement la balance des batailles entre le bien et le mal.";
		next;
		mes "[Gypsie]";
		mes "Souhaitez-vous devenir Gypsie?";
		next;
		if (select("Non.:Oui.") == 1) {
			mes "[Gypsie]";
			mes "Revenez quand vous vous sentirez prête.";
			next;
			mes "[Gypsie]";
			mes "Honneur aux Guerriers!";
			close;
		}
		if (SkillPoint) {
			mes "[Gypsie]";
			mes "Vous n'avez pas terminé votre apprentissage.";
			close;
		}
		jobchange Job_Gypsy;
		set ADVJOB,0;
		mes "[Gypsie]";
		mes "Félicitations!";
		mes "En tant que Gypsie, je sais que vos performances feront balancer les coeurs de tous ceux qui vous regarderont danser...";
		close;
	}
	mes "[Gypsie]";
	mes "Bienvenue au Valhalla, le Hall des Honneurs.";
	next;
	mes "[Gypsie]";
	mes "Faîtes comme chez vous.";
	mes "Honneur aux Guerriers!";
	close;
}
