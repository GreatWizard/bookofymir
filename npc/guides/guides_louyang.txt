louyang,213,213,3	script	Soldat#BA::LouGuide	825,{
	mes "[Soldat]";
	mes "Bienvenue à Louyang, Cité à l'histoire ancienne où les héros des contes sont emplis de bravoure.";
	next;
	mes "[Soldat]";

	next;

	next;
	mes "[Soldat]";
	mes "Prenez votre temps pour visiter la ville. Nous espérons que vous apprécierez votre séjour à Louyang.";
	next;
	switch(select("Guide de la Ville:Effacer les Informations de la Mini-Carte:Annuler")) {
	case 1:
		mes "[Soldat]";
		mes "Quel lieu vous intéresse?";
		next;
		switch(select("Château Dragon:Cabinet du Médecin:Mairie:Armurerie:Boutique:Taverne:Annuler")) {
		case 1:
			mes "[Soldat]";
			mes "Le Château Dragon est situé au ^ff0000+^000000 indiqué actuellement sur votre Mini-Carte.";
			mes "C'est là que demeurent tous les nobles, y compris notre Seigneur.";
			next;
			mes "[Soldat]";
			mes "Puisque vous êtes un visiteur, vous devriez aller saluer notre Seigneur à votre arrivée en ville.";
			viewpoint 1,218,255,2,0xFF3355;
			break;
		case 2:
			mes "[Soldat]";
			mes "Notre Médecin est très réputé.";
			mes "Vous trouverez son cabinet au ^8a0000+^000000 indiqué actuellement sur votre Mini-Carte.";
			next;
			mes "[Soldat]";
			mes "Aucune maladie n'est incurable pour notre Médecin!";
			mes "Enfin je ne suis pas allé vérifier...";
			viewpoint 1,263,94,3,0xCE6300;
			break;
		case 3:
			mes "[Soldat]";
			mes "La Mairie est le lieu où siège le Gouvernement Fédéral.";
			mes "Elle est située au ^777777+^000000 indiqué actuellement sur votre Mini-Carte.";
			next;
			mes "[Soldat]";
			mes "Si vous avez quelque problème que ce soit, vous devriez parler aux employés de la Mairie.";
			viewpoint 1,309,80,4,0xA5BAAD;
			break;
		case 4:
			mes "[Soldat]";
			mes "L'Armurerie est située au ^00ff00+^000000 indiqué actuellement sur votre Mini-Carte.";
			next;
			mes "[Soldat]";
			mes "De merveilleuses armes y sont exposées, des trésors créés par les forgerons les plus expérimentés de Louyang!";
			viewpoint 1,145,174,5,0x55FF33;
			break;
		case 5:
			mes "[Soldat]";
			mes "La Boutique est située au ^0000ff+^000000 indiqué actuellement sur votre Mini-Carte.";
			next;
			mes "[Soldat]";
			mes "Connais ton ennemi et connais toi toi-même!";
			mes "Pourquoi n'iriez-vous pas jeter un oeil et qui sait, trouver des petits plus qui feront la différence dans une bataille?";
			viewpoint 1,135,98,6,0x3355FF;
			break;
		case 6:
			mes "[Soldat]";
			mes "Si vous souhaitez vous reposer ou vous changer les idées, la Taverne est le lieu idéal!";
			mes "Elle est située au ^00ff00+^000000 indiqué actuellement sur votre Mini-Carte.";
			next;
			mes "[Soldat]";
			mes "La Taverne est également un excellent lieu pour rencontrer d'autres touristes, ainsi que pour glaner quelques informations...";
			viewpoint 1,280,167,7,0x00FF00;
			break;
		case 7:
			mes "[Soldat]";
			mes "N'hésitez pas à me demander si vous voulez que j'efface les informations de votre Mini-Carte.";
		}
		break;
	case 2:
		viewpoint 2,218,255,2,0xFF00FF;
		viewpoint 2,263,94,3,0xFF00FF;
		viewpoint 2,309,80,4,0xFF00FF;
		viewpoint 2,145,174,5,0xFF00FF;
		viewpoint 2,135,98,6,0xFF00FF;
		viewpoint 2,280,167,7,0xFF00FF;
		mes "[Soldat]";
		mes "Toutes les informations ont été effacées! N'hésitez pas à vous adresser à moi si vous avez besoin de quelque chose.";
		break;
	case 3:
		mes "[Soldat]";
		mes "Je comprends, vous souhaitez explorer Louyang et ses alentours par vous-même. Très bien, bonne journée!";
	}
	close;
}
