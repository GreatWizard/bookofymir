y_airport,126,43,4	script	Hôtesse#001::AirportY	90,{
	mes "[Personnel de l'Aéroport]";
	mes "Bienvenue à l'Aéroport.";
	mes "En quoi puis-je vous aider?";
	next;
	if (select("Embarquer:Annuler") == 1) {
		mes "Les frais d'embarquement sont de 1200 Zenys, mais si vous avez obtenu un billet gratuit vous n'aurez pas à payer ces frais. Voulez-vous prendre le zeppelin?";
		next;
		if (select("Oui:Non") == 1) {
			if(countitem(7311) > 0) {
				delitem 7311,1;
				warp "y_airport",148,51;
				close;
			}
			if(zeny >= 1200) {
				set Zeny,zeny-1200;
				warp "y_airport",148,51;
				close;
			}
			mes "[Personnel de l'Aéroport]";
			mes "Vous n'avez pas assez d'argent.";
			close;
		}
	}
	mes "[Personnel de l'Aéroport]";
	mes "Merci. Passez une agréable journée!";
	close;
}

y_airport,126,51,4	script	Hôtesse#004::AirportY2	90,{
	mes "[Personnel de l'Aéroport]";
	mes "Bienvenue~";
	mes "Veuillez suivre ce chemin afin d'embarquer sur le Zeppelin.";
	next;
	if (select("Retourner au Terminal Principal:Annuler") == 1) {
		mes "[Personnel de l'Aéroport]";
		mes "Si vous quittez le Terminal Principal, vous devrez à nouveau payer les frais d'embarquement.";
		mes "Voulez-vous vraiment sortir?";
		next;
		if (select("Oui:Non") == 1) {
			warp "y_airport",142,40;
			close;
		}
	}
	mes "[Personnel de l'Aéroport]";
	mes "D'accord. Merci de votre confiance. Nous espérons que vous passerez un agréable voyage~";
	close;
}
