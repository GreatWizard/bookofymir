S_SellSets:
	set .@item_cost,getiteminfo(getarg(0),0);
	set .@item_weight,getiteminfo(getarg(0),6);
	mes "[Madeleine Chu]";
	mes "Combien de Nécessaire de Cuisine " + (getarg(0)==12125 ? "Portatifs":"") + " souhaitez-vous acheter?";
	mes "Si vous changez d'avis, veuillez entrer 0.";
	next;
	while(1) {
		input .@sell;
		if (.@sell == 0) {
			mes "[Madeleine Chu]";
			mes "Vous avez changé d'avis?";
			mes "Bien, si vous avez besoin d'un autre "+getitemname(getarg(0))+" plus tard, vous pouvez venir me le demander à tout moment, d'accord?";
			close;
		}
		else if (.@sell > 100) {
			mes "[Madeleine Chu]";
			mes "Oh, je suis désolée mais je ne peux vendre plus de 100 "+getitemname(getarg(0))+" à chaque fois.";
			next;
		}
		else break;
	}
	set .@total_cost,.@sell * .@item_cost;
	set .@total_weight,.@sell * .@item_weight;
	if (zeny < .@total_cost) {
		mes "[Madeleine Chu]";
		mes "Oh, je suis désolée mais vous ne pouvez pas vous permettre d'acheter autant de ces "+getitemname(getarg(0))+".";
		mes "Veuillez apporter la somme exacte en zénys si vous voulez autant de Kits~";
		close;
	}
	if (!checkweight(getarg(0),.@sell)) {
		mes "[Madeleine Chu]";
		mes "Je suis désolée mais vous n'avez pas assez de place dans votre inventaire pour emporter autant de "+getitemname(getarg(0))+"...";
		close;
	}
	set zeny,zeny-.@total_cost;
	getitem getarg(0),.@sell;
	mes "[Madeleine Chu]";
	mes "Vous voici~";
	mes "Que la chance soit avec vous pendant votre formation culinaire!";
	return;
}
