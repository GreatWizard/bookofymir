		mes "Si vous voulez aller à Amatsu, faites-moi signe...!";

		mes "[Walter Moers]";
		mes "Oui, Amatsu est un endroit super!";
		mes "Mais l'océan peut se révéler traître, ce n'est pas facile de le traverser.";
		mes "J'espère que le prix du voyage vous convient.";

		mes "Le voyage coûte 10 000 Zenys.";
		mes "Nos navires de commerce sont robustes!";

		mes "Alors, prêt à partir?";
		mes "Avez-vous assez d'argent pour le voyage?";

			if (Zeny > 9999) {
				mes "[Walter Moers]";
				mes "Tout le monde à bord!";
				close2;
				set zeny,zeny-10000;
				warp "amatsu",197,83;
				end;
			}
			mes "[Walter Moers]";
				mes "Avez-vous oublié ce que j'ai dit?";
				mes "C'est 10 000 Zenys. Si vous ne pouvez pas me payer, vous n'avez qu'à aller faire un tour dans le bateau échoué et espérer que la chance soit avec vous durant votre chasse au trésor...";

		}
		mes "[Walter Moers]";
		mes "Si vous êtes un véritable aventurier,";
		mes "vous devriez visiter d'autres continents.";
		mes "Il n'y a rien de plus excitant que de découvrir de nouvelles cultures...";
		mes "Enfin, vous faites ce que vous voulez!";
		close;
	case 3:
		mes "[Walter Moers]";
		mes "Pas attiré par ce nouveau continent?";
		mes "Les gens n'aiment pas beaucoup s'aventurer trop loin de la côte.";
