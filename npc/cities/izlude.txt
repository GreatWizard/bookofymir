izlude,201,181,2	script	Marin#izlude	100,{
	mes "[Marin]";
	mes "Holà tout le monde, votre attention s'il vous plaît! Venez voyager et défier le Vent sur un bateau fantastique!!! Allez! Dépêchons!";
	next;
	switch(select("Ile de Byalan -> 150 Zenys:Marina d'Alberta -> 500 Zenys:Annuler")) {
	case 1:
		if (zeny < 150) {
			mes "[Marin]";
			mes "Vous n'avez pas assez d'argent!";
			close;
		}
		set zeny,zeny-150;
		warp "izlu2dun",107,50;
		end;
	case 2:
		if (zeny < 500) {
			mes "[Marin]";
			mes "Vous n'avez pas assez d'argent!";
			close;
		}
		set zeny,zeny-500;
		warp "alberta",188,169;
		end;
	case 3:
		close;
	}
}
