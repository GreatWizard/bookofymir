		mes "Voulez-vous embarquer maintenant?";
		next;

				set zeny,zeny-10000;
				warp "gon_fild01",258,82;
				end;
			}
			mes "[Wa Bai Hu]";
			mes "Je suis désolé mais vous devez avoir 10.000 zenys pour embarquer pour Gonryun. Veuillez d'abord vérifier que vous avez cette somme sur vous. Merci de votre intérêt. A bientôt.";
			close;
		}
		mes "[Wa Bai Hu]";
		mes "Je vois. Cependant, n'hésitez pas à revenir me voir si vous changez d'avis. Ce serait pour un grand honneur de pouvoir aider un vénérable étranger tel que vous.";
		close;
	case 3:
		mes "[Wa Bai Hu]";
		mes "Je vois. Cependant, n'hésitez pas à revenir me voir si vous changez d'avis. Ce serait pour un grand honneur de pouvoir aider un vénérable étranger tel que vous.";
