prontera,146,232,4	script	Hypnotiseur#2	124,{
	mes "[Hypnotiseur]";
	mes "Salutations, aventurier.";
	mes "Je suis un membre de l'Académie d'Hypnotisation en détachement à Prontera, afin de proposer nos services de Reset Skill à ceux qui le souhaiteraient et à un bon prix, c'est à dire gratuitement!";
	next;
	mes "[Hypnotiseur]";
	mes "Bien que le nombre de Reset Skill est illimité, deux conditions doivent être cependant requises.";
	mes "Tout d'abord, vous devez posséder un ^FF0000Niveau de Base inférieur à 40^000000.";
	next;
	mes "[Hypnotiseur]";
	mes "Deuxièmement, seules les personnes exerçant le métier de";
	mes "^FF0000Swordman, Acolyte, Mage,";
	mes "Voleur, Archer, Marchand,";
	mes "Taekwon Boy, Taekwon Girl, Gunslinger ou Ninja^000000";
	mes "peuvent bénéficier de ce service.";
	mes "Avez-vous des questions?";
	next;
	switch(select("Qu'est-ce qu'un Reset Skill?:Je souhaite faire un Reset Skill.:Rien, merci.")) {

	case 1:
		mes "[Hypnotiseur]";
		mes "Le service de Reset Skill permet aux aventuriers de redistribuer leurs Points de Compétences s'ils ne sont pas satisfaits de leurs compétences actuelles.";
		next;
		mes "[Hypnotiseur]";
		mes "Avant de procéder à un Reset Skill, le poids de l'inventaire doit être réduit à 0, par exemple en déposant les objets dans le Stock Kafra.";
		next;
		mes "[Hypnotiseur]";
		mes "Ah, si vous êtes marchand il vous faudra également retirer votre ^FF0000Charrette^000000.";
		close;

	case 2:
		mes "[Hypnotiseur]";
		mes "Êtes-vous certain de vouloir utiliser notre service de";
		mes "^FF0000Reset Skill^000000?";
		next;
		if (select("Oui:Non") == 1) {
			if (BaseLevel > 40) {
				mes "[Hypnotiseur]";
				mes "Je suis désolé, mais votre Niveau de Base est supérieur à 40, ce qui est la limite pour pouvoir profiter du service de Reset Skill.";
				close;
			}
			if ((Class > 0 && Class < 7 && Upper == 0) || (Class == Job_Gunslinger) || (Class == Job_Ninja) || (Class == Job_Taekwon)) {
				if (checkcart()) {
					mes "[Hypnotiseur]";
					mes "Oh! Merci de retirez votre Charrette avant de procéder au Reset Skill.";
					mes "Merci de votre coopération~";
					close;
				}
				if (Weight) {
					mes "[Hypnotiseur]";
					mes "Afin de pouvoir procéder au Reset Skill, je vous demanderais de bien vouloir stocker les objets de votre inventaire dans votre Stock Kafra. Le poids de votre inventaire doit être ^FF0000égal à 0^000000.";
					next;
					mes "[Hypnotiseur]";
					mes "Merci de votre compréhension.";
					close;
				}
				mes "[Hypnotiseur]";
				mes "Merci d'avoir utilisé nos services. Veuillez à redistribuer vos points de compétences de façon réfléchie.";
				mes "Bonne chance dans vos aventures. Faites attention à vous au cours de vos voyages~";
				ResetSkill;
				close;
			}
			mes "[Hypnotiseur]";
			mes "Je suis désolé, mais votre profession ne permet pas l'accès à nos services de Reset Skill.";
			next;
			mes "[Hypnotiseur]";
			mes "seules les personnes exerçant le métier de";
			mes "^FF0000Swordman, Acolyte,";
			mes "Mage, Voleur, Archer,";
			mes "Marchand, Taekwon";
			mes "Boy, Taekwon Girl, Gunslinger et Ninja^000000";
			mes "peuvent bénéficier de ce service.";
			mes "Merci de votre compréhension.";
			close;
		}
		mes "[Hypnotiseur]";
		mes "Merci et bonne chance dans vos aventures. Faites attention à vous au cours de vos voyages~";
		close;

	case 3:
		mes "[Hypnotiseur]";
		mes "Merci et bonne chance dans vos aventures. Faites attention à vous au cours de vos voyages~";
		close;
	}

}
