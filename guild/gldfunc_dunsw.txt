function	script	F_GldDunSw	{
	set .@GID, GetCastleData(getarg(0),1);
	if (.@GID == 0) {
		mes "[ Voix Mystérieuse ]";
		mes " ' ..... Vous n'êtes pas assez brave pour pénétrer ici...... ' ";
		close;
	}
	else {
		mes "[ Voix Mystérieuse ]";
		mes " ' Seuls les braves peuvent franchir cette limite... '";
		next;
		mes " ";
		mes "(Vous remarquez un petit levier....... Voulez-vous l'activer?)";
		next;

		if (select("Activer.:Ne rien faire.") == 1) {
			if (getcharid(2) == .@GID) {
				warp "gld_dun"+getarg(1),getarg(2),getarg(3);
				end;
			}
			mes " ";
			mes " Rien ne se produit.";
		}
		return;
	}
}
