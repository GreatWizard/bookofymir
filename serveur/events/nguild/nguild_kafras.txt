function	script	F_GKafra	{
	cutin "kafra_01",2;
	set @GID, GetCastleData(getarg(0),1);
	if (getcharid(2) == @GID && getgdskilllv(@GID,10001)) goto L_StartG;

	mes "[Employée Kafra]";
	mes "Je suis employée ici afin de servir les membres de la Guilde ^5533FF" + GetGuildName(@GID) + "^000000 uniquement. Merci de vous adresser à un autre Agent de la Corporation Kafra.";
	cutin "",255;
	close;
L_StartG:
	set @wrpP[0], 200;
	set @wrpD$[0], getarg(1);
	setarray @wrpC$[0], @wrpD$[0]+" ^880000"+@wrpP[0]+"^000000 z", "Annuler", "", "", "","";
	callfunc "F_Kafra",2,0,0,0,800;
	end;
}

// Castle 1 ===============================================
nguild_alde,218,170,0	script	Employée Kafra#N01	117,{
	callfunc "F_GKafra", "nguild_alde", "Prontera";
	end;
OnRecvCastleN01:
	if (GetCastleData("nguild_alde",9) < 1) disablenpc "Employée Kafra#N01";
	end;
}
