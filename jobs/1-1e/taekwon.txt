payon,157,141,5	script	Phoenix#TKJobChange	753,{

	if (Class == Job_Taekwon || Class == Job_Star_Gladiator || Class == Job_Soul_Linker)goto L_Complete;
	if (TK_Q == 1)goto L_Activated2;
	if (TK_Q == 2)goto L_JobChange;
	if (TAEK_Q > 0)goto L_Activated1;

	mes "[Phoenix]";
	mes "Cette terre... Ce monde magnifique a été souillé par le mal... Trop d'hommes se sont laissés corrompre par les ombres, trop de monstres menacent aujourd'hui les innocents...";
	next;
	mes "[Phoenix]";
	mes "Ce monde ravagé ne sera bientôt plus accueillant pour les humains, s'ils ne se dressent pas contre les forces obscures. Certains doivent se porter volontaires et combattre, pour la libération du monde!";
	next;
	mes "[Phoenix]";	
	mes "Je fais partie de ceux-là... J'ai choisi de m'entraîner sans relâche afin de devenir plus fort. J'ai développé un art du combat faisant appel au corps et à l'esprit, à l'endurance et à la puissance. Il est temps maintenant pour moi de transmettre cet art du combat aux jeunes courageux qui souhaitent purifier nos terres de l'influence maléfique des monstres.";
	next;
	mes "[Phoenix]";
	mes "Seul, je ne serai certainement pas capable de changer la face du monde... Mais je ne dois pas cesser de m'entraîner, physiquement et intellectuellement. Un jour viendra, où nous serons une armée...";
	next;
	mes "[Phoenix]";
	mes "Vous... Souhaitez-vous rejoindre la voie de l'entraînement afin que je vous enseigne mon art du combat? Ce sera long et douloureux, mais vous en ressortirez plus fort... assurement...";
	next;
	menu "Je souhaite rejoindre vos rangs.",L_Join,"Non, merci.",L_NoThx;

L_Join:
	if (Class != Job_Novice || JobLevel < 9) goto L_Failed1;
	mes "[Phoenix]";
	mes "Très bien... Je vous accepte en tant qu'élève. Au début de votre entraînement, votre corps devra apprendre à endurer l'effort.";
	next;
	mes "[Phoenix]";
	mes "L'esprit n'est pas forcément lié aux limites corporelles. Vous devez apprendre à utiliser votre corps et votre esprit ensemble, mais aussi séparément, afin d'atteindre votre plein potentiel.";
	mes "Retournez vous entraîner, et revenez me voir lorsque vous aurez ^ff00001 gagné plus d'endurance^000000...";
	next;
	mes "[Phoenix]";
	mes "Je sais que ce ne sera pas une tâche aisée, mais vous devez vous montrez capable de relever ce défi. J'espère que la prochaine fois que je vous reverrai, vous serez devenu plus fort.";
	set TAEK_Q,BaseLevel;
	close;

L_NoThx:
	mes "[Phoenix]";
	mes "Je comprends. Chacun décide de ses chemins de vie, et vous avez choisi de continuer votre voyage. J'espère que vous trouverez une voie qui éclairera le sens de votre vie. Bon voyage, aventurier.";
	close;

L_Failed1:
	if (Class != Job_Novice)goto L_Failed2;
	mes "[Phoenix]";
	mes "Malheureusement, il semblerait que vous n'êtes pas encore prêt à choisir définitivement l'orientation de votre vie. Continuez de vous entraîner, jeune Novice.";
	close;

L_Failed2: // CUSTOM MESSAGE
	mes "[Phoenix]";
	mes "Malheureusement, il semblerait que vous ayez déjà choisi l'orientation de votre vie...";
	close;

L_Activated1:
	if(TAEK_Q < BaseLevel) goto L_Finish;
	mes "[Phoenix]";
	mes "Revenez lorsque vous aurez gagné plus d'endurance. Ne négligez jamais votre entraînement. J'espère que la prochaine fois que je vous reverrai, vous serez devenu plus fort.";
	close;

L_Finish:
	if (SkillPoint > 0) goto L_Failed1;
	mes "[Phoenix]";
	mes "Je sens que vous avez gagné une plus grande puissance, mais aussi une plus grande paix intérieure. L'entraînement physique est terminé. Nous sommes donc prêt pour commencer l'entraînement spirituel.";
	set TK_Q,1;
	close;

L_Activated2:
	mes "[Phoenix]";
	mes "Pour votre entraînement spirituel, je vais vous poser une série de question afin de tester la réactivité et la noblesse de votre esprit.";
	mes "Détendez-vous et répondez avec honnêteté. Votre volonté et vos convictions sont votre force, ne l'oubliez jamais.";
	next;
	mes "[Phoenix]";
	mes "Lorsque vous vous retrouvez face à un problème de grande ampleur, quelle est votre réaction?";
	next;
	menu "Je l'affronte.",L_TKQST1,"J'essaye de l'éviter.",L_TKQST2,"J'essaye de l'analyser.",L_TKQST3;

L_TKQST1:
	mes "[Phoenix]";
	mes "Oui, c'est la réponse que j'attendais. Même si au premier abord il semble que vous ne puissiez résoudre un problème seul, une détermination suffisamment forte dégagera tous les obstacles que vous rencontrerez.";
	next;
	mes "[Phoenix]";
	mes "Même si au final vous échez, vous aurez appris de nombreuses choses de cette expérience. Un essai qui n'est pas mené à terme ne peut apporter de résultats bénéfiques.";
	next;
	mes "[Phoenix]";
	mes "Puisque vous comprenez l'importance de la volonté d'un chacun, nous allons passer à la question suivante.";
	next;
	mes "[Phoenix]";
	mes "Au cours de vos futurs voyages, vous ferez de nombreuses rencontres, et serez confronté à la diversité d'opinion et d'origines culturelles. Inévitablement, vous rencontrerez un jour une personne dont vous ne partagerez pas les idéaux.";
	next;
	mes "[Phoenix]";
	mes "De même, cette personne ne comprendra pas le chemin que vous avez choisi pour votre vie. Vos deux points de vue entreront en conflit... Que faîtes-vous?";
	next;
	menu "Insister *j'ai raison, c'est tout!*",L_TKQST4,"Ignorer le point de vue conflictuel",L_TKQST5,"Accepter les différences et tirer des leçons",L_TKQST6;

L_TKQST4:
	mes "[Phoenix]";
	mes "Il est important d'avoir son propre point de vue. Cependant, vous devez apprendre à accepter le fait que vous puissiez avoir tort, et qu'un point de vue différent a autant de mérite que le votre.";
	next;
	mes "[Phoenix]";
	mes "Il n'existe aucune réponse absolue, et la lumière de la vérité peut se cacher dans de nombreuses ombres. Tel est la loi de la nature. Imposer ses idées aux autres est une pratique d'oppression très grave.";
	next;
	mes "[Phoenix]";
	mes "Ne vous limitez pas à un seul point de vue, n'adhérez pas à une seule vérité. Méditez sur ces paroles, et revenez me voir lorsque vous les aurez comprises.";
	close;

L_TKQST5:
	mes "[Phoenix]";
	mes "Il est important de bien s'entendre avec les autres, mais vous ne retirerez aucun profit dans ce monde si vous n'apportez pas votre contribution, vos idées, vos pensées, vos opinions.";
	next;
	mes "[Phoenix]";
	mes "Lorsqu'un conflit d'idéaux éclate, si la discussion est menée avec respect, il peut s'agit d'une grande opportunité afin d'ouvrir votre esprit, votre compréhension du monde, et des autres.";
	next;
	mes "[Phoenix]";
	mes "Je vous laisse méditer sur cette idée d'établir une harmonie entre les autres et vous-même. Revenez me voir quand vous y aurez réfléchi.";
	close;

L_TKQST6:
	mes "[Phoenix]";
	mes "Bien. Vous devez accepter les critiques avec une considération sincère.";
	next;
	mes "[Phoenix]";
	mes "Il est impossible de tout savoir dans ce monde. Il est impossible de comprendre tous les poins de vue. Mais cela ne signifie pas que ces points de vue n'ont pas à être pris en compte.";
	next;
	mes "[Phoenix]";
	mes "Celui qui est en désaccord avec vous a peut-être la réponse que vous cherchez. Dans des temps sombres, cette personne peut être d'un grand secours. Souvenez-vous en.";
	next;
	mes "Vos réponses sont tout à fait satisfaisantes. Je vous laisse méditer un instant sur notre discussion. Revenez me voir lorsque votre esprit sera plus au calme.";
	set TK_Q,2;
	close;

L_TKQST2:
	mes "[Phoenix]";
	mes "Imbécile! Comment espérez-vous grandir et évoluer si vous fuyez les défis?! La peur est parfois une réaction saine et peut vous sauvez la vie, mais la lâcheté est impardonnable.";
	next;
	mes "[Phoenix]";
	mes "Vous me décevez beaucoup. Ne dîtes plus jamais une chose pareille devant moi. Allez donc méditer sur la différence entre le courage et la lâcheté. Lorsque vous aurez compris la différence entre les deux, revenez me voir.";
	close;

L_TKQST3:
	mes "[Phoenix]";
	mes "L'analyser? Ce n'est pas votre rôle. Vous devez éliminer le problème le plus rapidement possible. Comment pourriez-vous l'analyser sans y avoir été confronté de prêt? De plus, vous n'aurez pas toujours le temps ou la possibilité d'effectuer une analyse...";
	next;
	mes "[Phoenix]";
	mes "Certains problèmes peuvent être prédits et analysés à l'avance. Seulement tout ceci reste de la théorie. Une retraite trop précoce face à un problème peut faire perdre ne nombreuses informations, et se révèle au final peu sage. Méditez sur vos peurs et sur l'hésitation qui vous enserre, et revenez me voir lorsque vous aurez réglé vos conflits intérieurs.";
	close;

L_JobChange:
	mes "[Phoenix]";
	mes "Sentez-vous ce calme, cette paix? ... Je vais vous poser une question très importante. Répondez moi avec honnêteté.";	
	next;
	mes "[Phoenix]";
	mes "^ff0000Êtes-vous prêt à vous consacrer corps et âme à l'apprentissage de l'Art que je vous enseignerai, et de vous entraîner sans relâche afin d'être digne de la philosophie de l'Art TaekwonDo?^000000";
	next;
	menu "Oui.",L_JobChangeA,"Non.",L_JobChangeB;

L_JobChangeA:
	mes "[Phoenix]";
	mes "Très bien. Vous devez maintenant prendre conscience du pouvoir et des responsabilités d'un disciple ^ff0000Taekwondo^000000.";
	next;
	mes "[Phoenix]";
	mes "'Taekwon' signifie le poing et le coup, et 'Do' signifie l'art. Cet art martial repose sur l'utilisation des pieds et des poings.";
	next;
	mes "[Phoenix]";
	mes "Entraînez-vous à l'Art du Taekwondo afin de développer votre corps et votre esprit. Ne reculez jamais devant l'effort, ou vous apporterez la honte sur les disciples TaekwonDo.";
	next;
	mes "[Phoenix]";
	mes "Vous êtes si jeune... et pourtant vous méritez déjà de devenir un disciple de l'Art du ^ff0000Taekwondo^000000.";
	callfunc "Job_Change",Job_Taekwon;
	callfunc "F_ClearJobVar";
	getitem 2101,1; // Guard[0]
	next;
	mes "[Phoenix]";
	mes "Prenez cet uniforme d'entraînement et ce bouclier. Faîtes bon usage de vos dons pour le combat. Au cours de vos voyages, éclairez les autres à propos de notre Art, et ne cessez jamais d'apprendre de nouvelles techniques.";
	next;
	mes "[Phoenix]";
	mes "Il est maintenant temps pour vous de partir en voyage afin de trouver de nouveaux défis et développer votre force. Soyez fier de pratiquer l'Art du Taekwondo...";
	next;
	mes "[Phoenix]";
	mes "Bonne chance. J'espère vous revoir un jour, "+strcharinfo(0)+".";
	close;

L_JobChangeB:
	mes "[Phoenix]";
	mes "Hum, vous n'êtes sûrement pas prêt à devenir un disciple à part entière. Quand vous vous sentirez prêt, revenez me voir.";
	close;

	mes "[Phoenix]";
	mes "Comment se passe votre entraînement? Tout en perfectionnant vos techniques, n'oubliez pas de conserver des bases solides.";
	close;
